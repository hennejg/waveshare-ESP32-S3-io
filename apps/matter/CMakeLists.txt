cmake_minimum_required(VERSION 3.16)

# Shared components — list explicitly so cmake never scans components/quickjs
# or components/scripting (neither is used by this app and quickjs requires its
# own submodule which is not initialized for the matter build).
set(EXTRA_COMPONENT_DIRS
    "${CMAKE_CURRENT_SOURCE_DIR}/../../components/board"
    "${CMAKE_CURRENT_SOURCE_DIR}/../../components/esp32-wifi-bootstrap"
)

# esp-matter is required for this app.
if(NOT EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/../../esp-matter/connectedhomeip/connectedhomeip/BUILD.gn")
    message(FATAL_ERROR "esp-matter submodule not initialised — run: git submodule update --init --recursive esp-matter")
endif()

get_filename_component(ESP_MATTER_PATH
    ${CMAKE_CURRENT_SOURCE_DIR}/../../esp-matter REALPATH)
set(ENV{ESP_MATTER_PATH} ${ESP_MATTER_PATH})

set(MATTER_SDK_PATH ${ESP_MATTER_PATH}/connectedhomeip/connectedhomeip)

set(_PW_ENV_ROOT "${MATTER_SDK_PATH}/.environment")
file(WRITE
    "${MATTER_SDK_PATH}/config/esp32/third_party/connectedhomeip/build_overrides/pigweed_environment.gni"
    "# Auto-generated by CMakeLists.txt — hardcoded path, no env var needed.\n"
    "import(\"${_PW_ENV_ROOT}/build_overrides/pigweed_environment.gni\")\n")

list(PREPEND CMAKE_PROGRAM_PATH "${_PW_ENV_ROOT}/cipd/packages/pigweed")

list(APPEND EXTRA_COMPONENT_DIRS
    "${MATTER_SDK_PATH}/config/esp32/components"
    "${ESP_MATTER_PATH}/components")

include($ENV{IDF_PATH}/tools/cmake/project.cmake)

include(${ESP_MATTER_PATH}/examples/common/cmake_common/components_include.cmake)

project(waveshare_esp32s3_matter)

# matter.h lives in main/ (app-layer API); expose it to the board component
# which calls matter_di_update() / matter_do_update() when GPIO state changes.
idf_component_get_property(board_lib board COMPONENT_LIB)
target_include_directories(${board_lib} PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/main")

idf_build_set_property(CXX_COMPILE_OPTIONS
    "-std=gnu++17;-Os;-DCHIP_HAVE_CONFIG_H;-Wno-overloaded-virtual" APPEND)
idf_build_set_property(C_COMPILE_OPTIONS "-Os" APPEND)
idf_build_set_property(COMPILE_OPTIONS
    "-Wno-format-nonliteral;-Wno-format-security" APPEND)
