cmake_minimum_required(VERSION 3.16)

# esp-matter is optional: only enabled when the nested connectedhomeip submodule
# is checked out.  Without it the build proceeds with CONFIG_APP_MATTER_ENABLE=n
# (e.g. in CI where only the esp-idf container image is available).
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/esp-matter/connectedhomeip/connectedhomeip/CMakeLists.txt")
    get_filename_component(ESP_MATTER_PATH
        ${CMAKE_CURRENT_SOURCE_DIR}/esp-matter REALPATH)
    set(ENV{ESP_MATTER_PATH} ${ESP_MATTER_PATH})

    set(MATTER_SDK_PATH
        ${ESP_MATTER_PATH}/connectedhomeip/connectedhomeip)

    # ── GN / Pigweed setup ─────────────────────────────────────────────────────
    #
    # Problem: pigweed_environment.gni is loaded via
    #   import(getenv("_PW_ACTUAL_ENVIRONMENT_ROOT") + "/build_overrides/…")
    # That env var is set by pigweed's bootstrap.sh, but idf.py build does not
    # source it.  A wrapper-script approach fails on incremental builds because
    # the ExternalProject configure stamp stays valid and the old chip/build.ninja
    # (with raw gn embedded) is reused without re-running gn gen.
    #
    # Fix: rewrite the bootstrap GNI with an absolute import so no env var is
    # ever needed, regardless of how or when idf.py is invoked.
    set(_PW_ENV_ROOT "${MATTER_SDK_PATH}/.environment")
    file(WRITE
        "${MATTER_SDK_PATH}/config/esp32/third_party/connectedhomeip/build_overrides/pigweed_environment.gni"
        "# Auto-generated by CMakeLists.txt — hardcoded path, no env var needed.\n"
        "import(\"${_PW_ENV_ROOT}/build_overrides/pigweed_environment.gni\")\n")

    # Make GN findable without a manual symlink: add the cipd packages/pigweed
    # directory (where install.sh places gn) to cmake's program search path.
    list(PREPEND CMAKE_PROGRAM_PATH "${_PW_ENV_ROOT}/cipd/packages/pigweed")

    # Component search paths: chip stack + esp_matter component only.
    # We deliberately exclude device_hal (DevKit LED/button HAL) because we
    # have our own hardware abstraction for the Waveshare board.
    set(EXTRA_COMPONENT_DIRS
        "${MATTER_SDK_PATH}/config/esp32/components"
        "${ESP_MATTER_PATH}/components")

    set(ESP_MATTER_AVAILABLE TRUE)
else()
    message(STATUS "esp-matter submodule not initialised — Matter support disabled")
    # Append a no-matter fragment so the generated sdkconfig gets APP_MATTER_ENABLE=n
    set(SDKCONFIG_DEFAULTS "sdkconfig.defaults;sdkconfig.nomatter")
    set(ESP_MATTER_AVAILABLE FALSE)
endif()

include($ENV{IDF_PATH}/tools/cmake/project.cmake)

if(ESP_MATTER_AVAILABLE)
    # Merge Matter sdkconfig defaults (handles c2 quirks, sets CHIP_HAVE_CONFIG_H)
    include(${ESP_MATTER_PATH}/examples/common/cmake_common/components_include.cmake)
endif()

project(waveshare_esp32s3)

idf_build_set_property(CXX_COMPILE_OPTIONS
    "-std=gnu++17;-Os;-DCHIP_HAVE_CONFIG_H;-Wno-overloaded-virtual" APPEND)
idf_build_set_property(C_COMPILE_OPTIONS "-Os" APPEND)
idf_build_set_property(COMPILE_OPTIONS
    "-Wno-format-nonliteral;-Wno-format-security" APPEND)
